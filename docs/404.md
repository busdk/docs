The page you’re looking for doesn’t exist at this address. It may have moved, the URL may be misspelled, or the link may be out of date.

<p class="text-small color-fg-muted">
  Requested path: <span id="requested-path"></span>
</p>

To return to the entry point go to <a id="entry-link" href="{{ "/" | relative_url }}">Index page</a>.

You may want to start browsing from the closest entry point: <span id="closest-entry">Index page</span>.

<script>
  (function () {
    var base = "{{ site.baseurl | default: '' }}";
    var path = window.location.pathname || "/";

    if (base && base !== "/" && path.indexOf(base + "/") === 0) {
      path = path.slice(base.length);
    } else if (base && path === base) {
      path = "/";
    }

    if (!path) {
      path = "/";
    }

    var sections = [
      {
        prefix: "/modules",
        name: "Modules",
        url: "{{ "/modules/" | relative_url }}"
      }
    ];

    var ordered = sections.slice().sort(function (a, b) {
      return b.prefix.length - a.prefix.length;
    });

    var match = null;
    for (var i = 0; i < ordered.length; i += 1) {
      var candidate = ordered[i];
      if (path === candidate.prefix || path.indexOf(candidate.prefix + "/") === 0) {
        match = candidate;
        break;
      }
    }

    var pathTarget = document.getElementById("requested-path");
    if (pathTarget) {
      pathTarget.textContent = path;
    }

    if (match) {
      var entryLink = document.getElementById("entry-link");
      var closestEntry = document.getElementById("closest-entry");

      if (entryLink) {
        entryLink.textContent = match.name + " page";
        entryLink.setAttribute("href", match.url);
      }

      if (closestEntry) {
        closestEntry.textContent = match.name;
      }
    }
  })();
</script>
