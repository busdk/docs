<script>
  (function () {
    var base = "{{ site.baseurl | default: '' }}";
    var path = window.location.pathname || "/";

    if (base && base !== "/" && path.indexOf(base + "/") === 0) {
      path = path.slice(base.length);
    } else if (base && path === base) {
      path = "/";
    }

    if (!path) {
      path = "/";
    }

    var sections = [
      {
        prefix: "/modules",
        name: "Modules",
        url: "{{ "/modules/" | relative_url }}"
      }
    ];

    var ordered = sections.slice().sort(function (a, b) {
      return b.prefix.length - a.prefix.length;
    });

    var match = null;
    for (var i = 0; i < ordered.length; i += 1) {
      var candidate = ordered[i];
      if (path === candidate.prefix || path.indexOf(candidate.prefix + "/") === 0) {
        match = candidate;
        break;
      }
    }

    var pathTarget = document.getElementById("requested-path");
    if (pathTarget) {
      pathTarget.textContent = path;
    }

    if (match) {
      var container = document.getElementById("suggested-section");
      var text = document.getElementById("suggested-text");
      var link = document.getElementById("suggested-link");

      if (container && text && link) {
        text.textContent = "This looks like it belongs to the " + match.name + " section.";
        link.textContent = "Go to " + match.name + " index";
        link.setAttribute("href", match.url);
        container.hidden = false;
      }
    }
  })();
</script>
